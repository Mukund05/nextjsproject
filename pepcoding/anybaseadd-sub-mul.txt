import java.util.*;
  
  public class Main{
  //adding anybase no
  public static void main(String[] args) {
      Scanner scn = new Scanner(System.in);
      int b = scn.nextInt();
      int n1 = scn.nextInt();
      int n2 = scn.nextInt();
      int g1=anybasetodecimal(n1,b);   //n1 to decimal
      int g2=anybasetodecimal(n2,b);   //n2 to decimal
      int num=g1+g2;                   //adding decimal value
      int h=decimaltoanybase(num,b);   //added value convert to base value
      System.out.println(h);           //print final no
   }
  public static int anybasetodecimal(int n1,int b){  
      int d=1;
      int num=0;
      while(n1>0){
          int rem=n1%10;
          n1=n1/10;
          
          num=num+rem*d;
          d=d*b;
      }
      return num;
  }
  public static int decimaltoanybase(int n1,int b){
      int d=1;
      int num=0;
      while(n1>0){
          int rem=n1%b;
          n1=n1/b;
          
          num=num+rem*d;
          d=d*10;
      }
      return num;
  }
   
  }
